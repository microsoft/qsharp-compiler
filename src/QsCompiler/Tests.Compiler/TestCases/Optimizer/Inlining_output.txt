namespace Microsoft.Quantum.Testing.Optimization.Inlining {
    
    operation f (q : Qubit, n : Int) : Unit {
        
        if (n == 0) {
            // Do nothing
        }
        elif (n == 1) {
            T(q);
        }
        else {
            f(q, n - 1);
            f(q, n - 2);
        }
    }
    
    
    operation T (q : Qubit) : Unit {
        body intrinsic;
    }
    
    
    operation Test (q : Qubit) : Unit {
        
        T(q);
        T(q);
        T(q);
        T(q);
        T(q);
    }
    
}
