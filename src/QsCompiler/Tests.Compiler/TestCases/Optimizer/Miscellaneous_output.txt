    Function pickM (p, d) : (Qubit => Result) {
        QsBody (p, d) {
            let twice = (LocalVariable d * 2.000000);
            if (LocalVariable twice > 1.000000) {
                if (LocalVariable p = PauliZ) {
                    return GlobalCallable op0;
                }
            }
            return GlobalCallable M;
        }
    }

        QsBody (q) {
            repeat {
                mutable res = [Zero, Zero];

            return One;
        }
